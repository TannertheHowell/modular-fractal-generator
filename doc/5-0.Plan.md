function PixelColorOrIndex(c, palette)
    z = complex(0, 0)
    MAX_ITERATIONS = 100
    if palette is not None then
        len = size of palette
        TWO = 2
        for iter in range(len) do
            z = z * z + c
            if abs(z) > TWO then
                z = TWO
                if iter >= len then
                    iter = len - 1
                end if
                return palette[iter]
            end if
        end for
    else
        len = MAX_ITERATIONS
        TWO = 2
        for iter in range(len) do
            z = z * z + c
            if abs(z) > TWO then
                z = TWO
                if iter == MAX_ITERATIONS then
                    iter = MAX_ITERATIONS - 1
                end if
                return iter
            end if
        end for
    end if
    return palette[iter] if palette is not None and iter < size of palette else iter
